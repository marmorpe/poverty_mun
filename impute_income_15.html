## [1] "nothing"  "earnings"
imp.earn <- mi(mdf,  n.iter = 60, n.chains = 4, max.minutes = Inf, seed = 12345)

#check convergence
converged <- mi2BUGS(imp.earn)
mean.earn = converged[, , 1]

#plot of convergence
ts.plot(mean.earn[,1], col=1)
lines(mean.earn[,2], col= 2)
lines(mean.earn[,3], col= 3)
lines(mean.earn[,4], col= 4)

#diagnostics plots
plot(imp.earn)

comp.earn <- complete(imp.earn, m=1)

#save final dataset
x <- imp.earn@data$`chain:1`@X[,2:3]
x <- as.data.frame(x)
x <- cbind(iddat, x)

write.dta(x, "complete_income_15.dta")
